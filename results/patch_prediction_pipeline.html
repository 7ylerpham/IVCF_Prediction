''' Split images to blocks '''
def to_patches(images):
    global PATCHES_COUNTS 
    PATCHES_COUNTS = view_as_blocks(images[0], BLOCK_SHAPE).shape
    
    image_patches = []
    for img in images:
        temp = view_as_blocks(img, BLOCK_SHAPE)
        image_patches.append(temp)
    image_patches = np.reshape(image_patches, (-1, BLOCK_SHAPE[0], BLOCK_SHAPE[1], BLOCK_SHAPE[2]))
    
    return image_patches

''' Stitches blocks back to image '''
def patches_to_img(patches):
    temp = np.reshape(patches, (PATCHES_COUNTS[0], PATCHES_COUNTS[1], PATCHES_COUNTS[2], BLOCK_SHAPE[0], BLOCK_SHAPE[1], BLOCK_SHAPE[2]))
    image = unpatchify(temp, IMG_SHAPE)
    
    return image

def cropScan2(scan):  
    col_size = scan.shape[1]
    row_size = scan.shape[2]

    
    newScan = scan[:,
                   int(CROP_RATE*col_size):int((1-CROP_RATE)*col_size),
                   int(CROP_RATE*row_size):int((1-CROP_RATE)*row_size),:]
    return newScan

import sklearn
def plot_roc(name, labels, predictions, **kwargs):
    fp, tp, _ = sklearn.metrics.roc_curve(labels, predictions)
    plt.plot(100*fp, 100*tp, label=name, linewidth=2, **kwargs)
    plt.xlabel('False positives [%]')
    plt.ylabel('True positives [%]')
    plt.xlim([-0.5,105])
    plt.ylim([-0.5,105])
    plt.grid(True)
    ax = plt.gca()
    ax.set_aspect('equal')

def plot_prc(name, labels, predictions, **kwargs):
    precision, recall, _ = sklearn.metrics.precision_recall_curve(labels, predictions)

    plt.plot(precision, recall, label=name, linewidth=2, **kwargs)
    plt.xlabel('Precision')
    plt.ylabel('Recall')
    plt.grid(True)
    ax = plt.gca()
    ax.set_aspect('equal')

# Plot an image and predicted mask given the min/max slice range
def plot_two(image, predicted, min_, max_, name):
    index = min_
    count = 1
    fig = plt.figure(figsize = (8, 300))
    for x in range(min_, (max_+1)):
        for y in range(1, 3):
            plt.subplot((max_+1), 2, count).axis("off")
            if count % 2 == 1:
                plt.title("Image Slice: " + str(index))
                plt.imshow(image[:,:,index], cmap='gray')
            else:
                plt.title("Predicted Mask Slice: " + str(index))
                plt.imshow(predicted[:,:,index], cmap='gray')
                index += 1
            count += 1
    plt.savefig(name)
    plt.close()
